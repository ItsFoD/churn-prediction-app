# Store old and new values for comparison
encoded_comparison = pd.DataFrame()

for col in categorical_cols:
    le = LabelEncoder()
    old_values = df_copy2[col].copy()
    df_copy2[col] = le.fit_transform(df_copy2[col])
    new_values = df_copy2[col]
    temp_df = pd.DataFrame({f"{col}_old": old_values, f"{col}_new": new_values})
    encoded_comparison = pd.concat([encoded_comparison, temp_df], axis=1)
